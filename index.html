📒 配置错题本

数据存储在 Supabase 云数据库，电脑手机自动同步。

  1. 注册 supabase.com 免费账号
  2. 创建新项目，记下 Project URL 和 anon key
  3. 进入 SQL Editor，运行下方 SQL 创建表
  4. 填入 URL 和 Key，点击保存
📋 点击展开 SQL（复制到 Supabase SQL Editor 运行） 
CREATE TABLE subjects (
  id SERIAL PRIMARY KEY,
  name TEXT UNIQUE NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE entries (
  id SERIAL PRIMARY KEY,
  subject TEXT NOT NULL,
  source TEXT DEFAULT '',
  error_reason TEXT DEFAULT '',
  screenshot TEXT DEFAULT '',
  answer_image TEXT DEFAULT '',
  answer_text TEXT DEFAULT '',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE subjects ENABLE ROW LEVEL SECURITY;
ALTER TABLE entries ENABLE ROW LEVEL SECURITY;

CREATE POLICY "allow_all_subjects" ON subjects FOR ALL USING (true);
CREATE POLICY "allow_all_entries" ON entries FOR ALL USING (true);
